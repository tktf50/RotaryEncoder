82 unsigned char process() {
83 // Grab state of input pins.
84 unsigned char pinstate = (digitalRead(dtPin) << 1) | digitalRead(clkPin);
85 // Determine new state from the pins and state table.
86 state = ttable[state & 0xf][pinstate];
87 // Return emit bits, ie the generated event.
